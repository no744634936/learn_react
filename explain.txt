router 里面的component都可以传一个props 参数。这是router给component的




1,网页过两秒钟后从About自动跳转Content。


const About=(props)=>{
    // console.log(props);

    //两秒钟后跳转到content
    setTimeout(()=>{
        props.history.push("/content")
    },2000);
    
    return(
        <div className="container">
            <h4 className="center">
                <p>
                    this is About page
                </p>
            </h4>
        </div>
    );
}
